Calendar Chart

var calendarChart = new bluewave.charts.CalendarChart(parent, {});
                
var chartConfig = {
    date: "Date",
    value: "Volume",
};
   
var csv = "Date,Volume\n";

var year = new Date().getFullYear();
var tomorrow = new Date("01/01/"+year);

for (let i=0; i < 180; i++){
    tomorrow.setDate(tomorrow.getDate() + 1);
    csv += tomorrow + "," + Math.random()*1000000000 + "\n";
}

var data = d3.csvParse(csv);
    
var calendarChart = new bluewave.charts.CalendarChart(parent, {});
calendarChart.update(chartConfig, data);
